### 2. TypeScript ä»£ç†æœåŠ¡ (ParserBridge)

#### æ–°å¢æ–‡ä»¶: `backend/src/parser/ParserBridge.ts`

```typescript
import { spawn } from "bun";
import { join } from "path";

interface ParseResult {
    symbols: Symbol[];
    error?: string;
}

interface Symbol {
    id: string;
    type: string;
    name: string;
    startLine: number;
    endLine: number;
}

export class ParserBridge {
    private parserPath: string;
    
    constructor() {
        // æŸ¥æ‰¾ Rust CLI å·¥å…·
        this.parserPath = this.findParserBinary();
    }
    
    private findParserBinary(): string {
        const candidates = [
            join(process.cwd(), "codeview-parser"),
            join(process.cwd(), "bin", "codeview-parser"),
            "/usr/local/bin/codeview-parser",
        ];
        
        for (const path of candidates) {
            if (Bun.file(path).size > 0) {
                return path;
            }
        }
        
        throw new Error("codeview-parser binary not found");
    }
    
    async parseFile(filePath: string, language: string): Promise<ParseResult> {
        const proc = spawn([
            this.parserPath,
            "symbols",
            filePath,
            "--lang", language,
            "--output", "json"
        ]);
        
        const output = await new Response(proc.stdout).text();
        const exitCode = await proc.exited;
        
        if (exitCode !== 0) {
            return { symbols: [], error: `Parser exited with code ${exitCode}` };
        }
        
        return JSON.parse(output);
    }
    
    async batchParse(files: Array<{path: string, language: string}>): Promise<ParseResult[]> {
        const input = JSON.stringify({ files });
        
        const proc = spawn([
            this.parserPath,
            "batch",
            "--input", "-"  // ä» stdin è¯»å–
        ], {
            stdin: "pipe"
        });
        
        proc.stdin.write(input);
        proc.stdin.end();
        
        const output = await new Response(proc.stdout).text();
        const result = JSON.parse(output);
        
        return result.results;
    }
}
```
